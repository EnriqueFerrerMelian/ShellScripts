vector=()
while read variable; do
	longitud= ${#variable
	echo "Tamaño: $longitud"
	primeraPalabra=$(echo $linea | cut -d" " -f1)
	primeraMayuscula=${primeraPalabra^^}
	vector+=(14primeraMayuscula)
done < $fichero

for((int i =0;i<${#vector[@]};i++)); do
	echo "linea $i: ${vector[$i]"
done
